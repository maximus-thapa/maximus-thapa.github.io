    Simple login

    Full Code Example

    1. Create Your Project Structure

      src/
├── components/
│   ├── Home.js
│   ├── Login.js
│   └── NotFound.js
├── routes/
│   ├── AppRoutes.js
│   └── ProtectedRoute.js
├── App.js
└── index.js

    2. Install Required Packages Make sure you have React Router and Ant Design installed:

      npm install react-router-dom antd

    3. Code for Each Component

      index.js
      import React from "react";
import ReactDOM from "react-dom/client";
import App from "./App";
import "antd/dist/antd.css"; // Import Ant Design styles

const root = ReactDOM.createRoot(document.getElementById("root"));
root.render(<App />);
      App.js
      import React, { useState } from "react";
import { BrowserRouter as Router } from "react-router-dom";
import AppRoutes from "./routes/AppRoutes";
import Navbar from "./components/Navbar"; // Import your Navbar component

const App = () => {
    const [isAuthenticated, setIsAuthenticated] = useState(false);

    const handleLogin = (status) => {
        setIsAuthenticated(status);
    };

    return (
        <Router>
            {isAuthenticated && <Navbar />} {/* Show Navbar only if authenticated */}
            <AppRoutes isAuthenticated={isAuthenticated} onLogin={handleLogin} />
        </Router>
    );
};

export default App;
      AppRoutes.js
      import { Routes, Route } from "react-router-dom";
import Home from "../components/Home";
import NotFound from "../components/NotFound";
import Login from "../components/Login";
import ProtectedRoute from "./ProtectedRoute";

const AppRoutes = ({ isAuthenticated, onLogin }) => {
    return (
        <Routes>
            <Route path="/login" element={<Login onLogin={onLogin} />} />
            <Route
                path="/home"
                element={
                <ProtectedRoute isAuthenticated={isAuthenticated}>
                    <Home />
                </ProtectedRoute>
                }
            />
            <Route path="*" element={<NotFound />} />
        </Routes>
    );
};

export default AppRoutes;
      ProtectedRoute.js
      import React from "react";
import { Navigate } from "react-router-dom";

const ProtectedRoute = ({ isAuthenticated, children }) => {
    return isAuthenticated ? children : <Navigate to="/login" />;
};

export default ProtectedRoute;
      Home.js
      import React from "react";

const Home = () => {
    return <h1>Welcome to the Home Page!</h1>;
};

export default Home;
      NotFound.js
      import React from "react";

const NotFound = () => {
    return <h1>404 - Page Not Found</h1>;
};

export default NotFound;
      Login.js
      import React, { useState } from "react";
import { Form, Input, Button, Card, message } from "antd";
import { useNavigate } from "react-router-dom";

const Login = ({ onLogin }) => {
    const [loading, setLoading] = useState(false);
    const navigate = useNavigate(); // Initialize useNavigate

    const onFinish = async (values) => {
        setLoading(true);
        try {
            // Simulate an API call
            const response = await fakeLoginAPI(values);
            if (response.success) {
                onLogin(true); // Update the authentication state
                navigate("/home"); // Navigate to the home (dashboard) route
            } else {
                message.error(response.message);
            }
        } catch (error) {
            message.error("Login failed. Please try again.");
        } finally {
            setLoading(false);
        }
    };

    const fakeLoginAPI = async ({ username, password }) => {
        // Simulating an API call
        return new Promise((resolve) => {
            setTimeout(() => {
                if (username === "user" && password === "pass") {
                    resolve({ success: true });
                } else {
                    resolve({ success: false, message: "Invalid credentials" });
                }
            }, 1000);
        });
    };

    return (
        <Card title="Login" style={{ width: 300, margin: "auto", marginTop: 100 }}>
            <Form name="login" onFinish={onFinish}>
                <Form.Item name="username" rules={[{ required: true, message: 'Please input your username!' }]}>
                    <Input placeholder="Username" />
                </Form.Item>
                <Form.Item name="password" rules={[{ required: true, message: 'Please input your password!' }]}>
                    <Input.Password placeholder="Password" />
                </Form.Item>
                <Form.Item>
                    <Button type="primary" htmlType="submit" loading={loading} block>
                        Login
                    </Button>
                </Form.Item>
            </Form>
        </Card>
    );
};

export default Login;

    Summary

    • Login Component: Handles user authentication and simulates an API call.
    • Protected Route: Redirects to the login page if the user is not authenticated.
    • App Component: Manages the authentication state and routes.

    Running the Application

    1. Start your React application:

      npm start

    2. Navigate to http://localhost:3000/login to access the login page. Use username: user and password: pass to log in and access the home page.

      This setup provides a complete example of a React application with protected routing and a login flow using Ant Design and React Router. If you have any further questions or need additional features, feel free to ask!
