    Callback Function (कोलब्याक फङ्सन)

    कोलब्याक फङ्सन भनेको एउटा यस्तो फङ्सन हो जुन अर्को फङ्सनलाई आर्ग्युमेन्टको रूपमा पास गरिन्छ र त्यो फङ्सनले कुनै कार्य समाप्त भएपछि यसलाई कल गर्छ। यसलाई सामान्यतया असिन्क्रोनस कार्यहरूमा प्रयोग गरिन्छ।

    function calculateSum(a, b, callback) {
    const sum = a + b; // गणना गर्छ
    callback(sum);     // कोलब्याक फङ्सनलाई परिणामसहित कल गर्छ
}

function printResult(result) {
    console.log("नतिजा हो:", result);
}

// प्रयोग गर्दै
calculateSum(5, 3, printResult);

    Inner Function (इनर फङ्सन)

    इनर फङ्सन भनेको अर्को फङ्सन भित्र परिभाषित गरिएको फङ्सन हो। यसले बाहिरी फङ्सनका भेरिएबल र स्कोपमा पहुँच गर्न सक्छ।

    function outerFunction() {
    const greeting = "नमस्कार";

    function innerFunction() {
        console.log(greeting); // बाहिरी फङ्सनको भेरिएबल पहुँच गर्छ
    }

    innerFunction(); // इनर फङ्सनलाई कल गर्छ
}

// प्रयोग गर्दै
outerFunction();
    1. कोडको विभाजन: कोलब्याक र इनर फङ्सनले कोडलाई धेरै सफा र मर्मत गर्न सजिलो बनाउँछ।
    2. असिन्क्रोनस कार्यहरू: असिन्क्रोनस कार्यहरू जस्तै API कलहरूमा परिणामको ह्यान्डलिङका लागि कोलब्याक फङ्सन महत्त्वपूर्ण हुन्छ।
    3. स्कोप: इनर फङ्सनले बाहिरी फङ्सनको भेरिएबलहरूमा पहुँच गर्न सक्छ, जसले कोडको लचिलोपन र व्यावहारिकता बढाउँछ।

    यी कारणले गर्दा कोलब्याक र इनर फङ्सनहरू जाभास्क्रिप्टमा महत्त्वपूर्ण र उपयोगी छन्।

    फङ्सनलाई अर्को फङ्सनको आर्ग्युमेन्टको रूपमा पास गर्नुपर्ने कारणहरू तल व्याख्या गरिएका छन्:

    1. कार्यको विभाजन (Separation of Concerns)

      जब हामी फङ्सनलाई आर्ग्युमेन्टको रूपमा पास गर्छौं, यसले कार्यहरूलाई अलग गर्ने मौका दिन्छ। यसबाट मुख्य कार्य र त्यसपछि हुने कार्य अलग राख्न सकिन्छ। उदाहरणका लागि, गणना गर्ने फङ्सन र परिणामलाई प्रदर्शन गर्ने फङ्सनलाई छुट्टै राख्न सकिन्छ।

    2. लचिलोपन (Flexibility)

      जब हामी फङ्सनलाई आर्ग्युमेन्टको रूपमा पास गर्छौं, त्यसको अर्थ हामी विभिन्न फङ्सनहरू प्रयोग गरेर सोही मूल फङ्सनको व्यवहार परिवर्तन गर्न सक्छौं। यसले हामीलाई आवश्यकता अनुसार विभिन्न ह्यान्डलरहरू बनाउने अवसर दिन्छ।

    3. असिन्क्रोनस कार्यहरूमा प्रयोग (Usage in Asynchronous Operations)

      जाभास्क्रिप्टमा धेरै कार्यहरू असिन्क्रोनस हुन्छन्, जस्तै API कलहरू। जब कुनै कार्य समाप्त हुन्छ, त्यसपछि नतिजा ह्यान्डल गर्नका लागि कोलब्याक फङ्सनको प्रयोग गर्न सकिन्छ। यसले नतिजा पाएका बेला कुनै विशेष कार्य गर्न मद्दत गर्छ।

    4. स्कोपको पहुँच (Access to Scope)

      इनर फङ्सनहरूले बाहिरी फङ्सनका भेरिएबलहरूमा पहुँच गर्न सक्छन्। यसले फङ्सनभित्रको डेटा र स्थिति सुरक्षित राख्न मद्दत गर्छ। यसबाट कोडले ठूलो लचिलोपन प्राप्त गर्छ।

    function fetchData(callback) {
    setTimeout(() => {
        const data = { id: 1, name: "राम" };
        callback(data);
    }, 2000);
}

function handleData(data) {
    console.log("डेटा प्राप्त भयो:", data);
}

// यहाँ fetchData लाई handleData को रूपमा पास गरिएको छ
fetchData(handleData);

    यसैले, फङ्सनलाई अर्को फङ्सनको आर्ग्युमेन्टको रूपमा पास गर्नु महत्त्वपूर्ण छ। यसले कोडलाई मर्मत गर्न सजिलो, लचिलो र असिन्क्रोनस कार्यहरूमा प्रबन्ध गर्न सक्षम बनाउँछ।

    Alternative to callback function

    what are the advance alternative to callback function. explain in nepali

    कोलब्याक फङ्सनहरूको सट्टामा प्रयोग गर्न सकिने केही उन्नत विकल्पहरू निम्नलिखित छन्:

    1. प्रमिस (Promises)

      प्रमिसले असिन्क्रोनस कार्यहरूलाई सजिलो र प्रबन्ध गर्न मिल्ने बनाउँछ। यसले दुई मुख्य अवस्थाहरूको प्रबन्ध गर्दछ: "fulfilled" (सफलता) र "rejected" (अस्वीकृति)।

      उदाहरण
      function fetchData() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            const data = { id: 1, name: "राम" };
            resolve(data); // सफलतासाथ डेटा पठाउँछ
        }, 2000);
    });
}

fetchData()
    .then(data => {
        console.log("डेटा प्राप्त भयो:", data);
    })
    .catch(error => {
        console.log("त्रुटि:", error);
    });

    2. Async/Await

      Async/Await प्रमिसहरूको अधिक सजिलो र पठनीयता भएको तरिका हो। यसले असिन्क्रोनस कोडलाई समान्य कोड जस्तै लेख्न अनुमति दिन्छ।

      उदाहरण
      async function getData() {
    try {
        const data = await fetchData(); // प्रमिसको परिणामको प्रतीक्षा गर्छ
        console.log("डेटा प्राप्त भयो:", data);
    } catch (error) {
        console.log("त्रुटि:", error);
    }
}

getData();

    3. Observables (RxJS)

      Observablesले डेटा स्ट्रीमहरूलाई प्रबन्ध गर्नको लागि प्रयोग गरिन्छ। यसले तपाईंलाई विभिन्न घटनाहरूको सुचना प्राप्त गर्न अनुमति दिन्छ र धेरै शक्तिशाली र लचिलो बनाउँछ।

      उदाहरण
      import { Observable } from 'rxjs';

const dataObservable = new Observable(subscriber => {
    setTimeout(() => {
        subscriber.next({ id: 1, name: "राम" });
        subscriber.complete(); // समाप्त
    }, 2000);
});

dataObservable.subscribe({
    next(data) {
        console.log("डेटा प्राप्त भयो:", data);
    },
    complete() {
        console.log("डेटा प्रक्रिया समाप्त भयो");
    }
});

    4. Event Emitter

      Event Emitterहरूले इभेन्ट ड्रिभन प्रोग्रामिङको लागि उपयुक्त हुन्छ। यसले इभेन्टहरूलाई सुन्न र तिनीहरूलाई प्रबन्ध गर्न अनुमति दिन्छ।

      उदाहरण
      const EventEmitter = require('events');
const myEmitter = new EventEmitter();

myEmitter.on('dataReceived', (data) => {
    console.log("डेटा प्राप्त भयो:", data);
});

// डेटा प्राप्त गर्नको लागि इभेन्टलाई ट्रिगर गर्छ
setTimeout(() => {
    myEmitter.emit('dataReceived', { id: 1, name: "राम" });
}, 2000);

    निष्कर्ष

    यी विकल्पहरूले कोलब्याक फङ्सनहरूको सीमाहरू र जटिलताहरूलाई समेट्न मद्दत गर्छ। प्रमिस, Async/Await, Observables, र Event Emittersले असिन्क्रोनस कार्यहरूको प्रबन्धलाई सजिलो, पठनीय र लचिलो बनाउँछ।
